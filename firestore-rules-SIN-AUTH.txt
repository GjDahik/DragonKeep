// ============================================================
// REGLAS PARA PEGAR EN FIREBASE CONSOLE → Firestore → Rules
// Funcionan SIN cambiar la app (sin Auth anónimo ni sessions).
// La app funcionará tal como está; la seguridad depende de no
// compartir el enlace y de uso privado. Para más seguridad,
// más adelante podés volver a activar Auth + sessions en la app
// y usar las reglas de firestore.rules.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    match /dms/{dmId} {
      allow read, write: if true;
    }

    match /players/{playerId} {
      allow read, write: if true;
    }

    match /cities/{cityId} {
      allow read, write: if true;
      match /playerNotes/{noteId} {
        allow read, write: if true;
      }
    }

    match /npcs/{npcId} {
      allow read, write: if true;
    }

    match /shops/{shopId} {
      allow read, write: if true;
    }

    match /travelingShops/{shopId} {
      allow read, write: if true;
    }

    match /transactions/{txId} {
      allow read, write: if true;
    }

    match /notifications/{notifId} {
      allow read, write: if true;
    }

    match /automation_rules/{ruleId} {
      allow read, write: if true;
    }

    match /settings/{docId} {
      allow read, write: if true;
    }

    match /missions/{missionId} {
      allow read, write: if true;
    }

    match /legend_audio/{trackId} {
      allow read, write: if true;
    }

    match /rutas_conocidas/{rutaId} {
      allow read, write: if true;
    }

    match /bitacora_viajes/{viajeId} {
      allow read, write: if true;
    }

    match /map_markers/{markerId} {
      allow read, write: if true;
    }

    match /player_map_markers/{playerId} {
      allow read, write: if true;
    }

    match /sessions/{uid} {
      allow read, write: if true;
    }
  }
}
